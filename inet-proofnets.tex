% ====================================================
% NETS
% ====================================================

\newcommand{\context}[4][]{
	% 1 = style
	% 2 = net of reference
	% 3 = node name
	% 4 = label
	\begin{pgfonlayer}{bg}
		\node [#1] (#3) {#4};
		\draw[doublelink] (#2) -- (#3);
	\end{pgfonlayer}
}

\newcommand{\straightcontext}[4][]{
	% 1 = style
	% 2 = net of reference
	% 3 = node name
	% 4 = label
	\begin{pgfonlayer}{bg}
		\node[below=5mm of #2] [#1] (#3) {#4};
		\draw[doublelink] (#2) -- (#3);
	\end{pgfonlayer}
}
