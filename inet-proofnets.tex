\newcommand{\promobox}[3][]{
	% 1 = style
	% 2 = name
	% 3 = content
	\inetbox[#1]{#2}{#3}
	\node[draw,circle,fill=white] at ($(#2.south west)!0.5!(#2.south east)$) (#2_sym) {$\bang$};
}
